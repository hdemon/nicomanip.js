545 			 * max、minのいずれか、もしくは両方を指定した場合、ソート結果からこれらの条件に従って抽出されたマイリストオブジェクトを返す。
546 			 * これらのパラメータを指定しない場合、マイリストID、アイテムID、ソート対象の値を要素とするオブジェクトの配列を返す。
547 			 * @memberOf MyNico.prototype
548  			 * @param {Object[]} argsObj.condition ソート条件オブジェクトを格納する配列。添字の小さい条件から適用される。 
549  			 * @param {String} argsObj.condition.name ニコニコ動画APIのパラメータに基づく、ソート対象のプロパティ名 
550  			 * @param {Boolean} argsObj.condition.ascend 昇順ならばtrue、降順ならばfalse 
551  			 * @param {Number} argsObj.max 上記パラメータのフィルタリング上限値 
552  			 * @param {Number} argsObj.min フィルタリング下限値
553  			 * @returns {Object} max,minの指定を行うことで、MyNico.mylistと同構造のオブジェクトを返し、チェーンメソッドを可能にする。
554  			 * @returns {Array} max,minのいずれも設定しない場合、マイリストID、アイテムID、ソート対象の値を要素とするオブジェクトの配列を返す。
555  			 * @example
556  			 * var mylist = new MyNico;
557  			 * mylist
558  			 *   .sort({
559 			 *     condition : [
560 			 *       {
561 			 *         name : "view_counter",
562 			 *         ascend : true
563 			 *       }					
564 			 *     ],
565 			 *     min : 10
566 			 *   })
567 			 *   .copy(123456789);
568 			 *
569 			 * 全動画／静画の内、閲覧数トップ10のみを含むオブジェクトを返し、
570 			 *     それをID:123456789のマイリストへコピーする。
571 			 */				
572 			sort : function( argsObj ){
573 				var array = [];
574 				argsObj.condition.forEach(function(e, i){
575 					var counter = 0;
576 					for (var mylistId in this.mylist){
577 						var mylist = this.mylist[mylistId];
578 						for (var itemId in mylist.elements){
579 							if (typeof array[counter] === "undefined") array[counter] = {};
580 							var el = mylist.elements[itemId];
581 							array[counter].value	= (
582 								typeof el === "String"
583 									? charCode(el[e.name])
584 									: el[e.name]
585 							);
586 							array[counter].itemId	= array[counter].itemId || itemId;
587 							array[counter].mylistId	= array[counter].mylistId || mylistId;
588 							counter++;
589 						}
590 					}
591 					mergeSort( array );
592 					if (!e.ascend) array.reverse();
593 				}.bind(this));
594 				
595 				if (
596 					typeof argsObj.high === "undefined" &&
597 					typeof argsObj.low === "undefined" 
598 				){
599 					return array;
600 				} else {
601 					var mylist = {},
602 						high = argsObj.high || 0,
603 						low = (typeof argsObj.low === "undefined" ? array.length : argsObj.low );
604 					for (var i=high; i<low; i++){
605 						var e = array[i],
606 							o = this.mylist[e.mylistId];							
607 						
608 						if (typeof mylist[e.mylistId] === "undefined") {
609 							mylist[e.mylistId] = {};
610 							var d = mylist[e.mylistId];
611 							extendDeep(o, d);
612 							delete d.elements;
613 							d.elements = {};
614 						}
615 						var d = mylist[e.mylistId];
616 						d.elements[e.itemId] = {};
617 						extendDeep(o.elements[e.itemId], d.elements[e.itemId]);
618 					}
619 					var duplication = Object.create(Object.getPrototypeOf(this));
620 					duplication.mylist = {};
621 					extendDeep(mylist, duplication.mylist);
622 					return duplication;
623 				}
624 			},

627 			 * APIに渡す"id_list"形式のパラメータを作る。APIが、コピー／移動先となるマイリストIDを１つずつしか受け付けないため、
628 			 * マイリストID毎に分割する処理を行う。また、とりあえずマイリストとの分離、文字列の長さを一定間隔に分割する等の
629 			 * 処理も行う。		 
630 			 * @memberOf MyNico.prototype
631 			 * @returns {String[]} list "id_list"形式の文字列を含む配列。序列はマイリストIDとアイテムIDの若さに従う。
632 			 */
633 			createIdList : function(){
634 				/*	1.mylistオブジェクトに含まれる全要素の個別リストを作成
635 					2.マイリストIDと、とりあえずマイリストとそれ以外を分けて、連結リストを作る。
636 					3.URLが一定の長さを超えた場合も分割する。				*/
637 
638 				var _list	= {},
639 					list	= [],
640 					counter = 0,
641 					num		= 0;
642 
643 				for (var mylistId in this.mylist){
644 					var mylist = this.mylist[mylistId];
645 					_list[mylistId] = _list[mylistId] || [];
646 
647 					for (var itemId in mylist.elements){
648 						_list[mylistId]
649 							.push(
650 								"&id_list[" +
651 								mylist.elements[itemId].item_type +
652 								"][]=" +
653 								itemId
654 							);
655 					}
656 				}
657 
658 				function concrete(_list, mylistId){
659 					var thisMylist = this.mylist[mylistId];
660 					if (thisMylist.type === "defList"){
661 						list[num] = { str : "" };
662 					} else if (thisMylist.type === "normalList") {
663 						list[num] = { str : "&group_id=" };
664 					}
665 					list[num].type = thisMylist.type;
666 
667 					// １つのリストが20~25Bなので、それに合わせてリスト連結数を決める。
668 					// 上限数に達したら、リストのカウントを１つ上げて続きを連結。
669 					for (var i=0, max = URLmax/20; i<max; i++){
670 						list[num].str += _list[mylistId].shift();
671 						if (_list[mylistId].length <= 0) {
672 							list[num].type = thisMylist.type;
673 							return;
674 						}
675 					}
676 					num++;
677 					concrete.bind(this)(_list, mylistId);
678 				}
679 
680 				for (var mylistId in _list){
681 					concrete.bind(this)(_list, mylistId);
682 					num++;
683 				}
684 
685 				return list;
686 			},
